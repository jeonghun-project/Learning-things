    while - break

    let count = 0;

while (true) {
    count++;
    
    if(count >= 20) {
        break;
    }
}

console.log(count); // 20

    while - do

    let count = 0;

do {
    count++;
    
    if(count >= 20) {
        break;
    }
}
while (false) {
 
}

console.log(count); // 1
