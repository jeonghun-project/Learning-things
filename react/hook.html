мҡ°лҰ¬лҠ” class componentлҘј нҶөн•ҳм—¬ react мғҒнғңкҙҖлҰ¬лҘј н•ҳлҠ” лІ•мқ„ мқөнһҲкі  мӮ¬мҡ©н•ҙліҙм•ҳм§Җл§Ң, class componentsк°Җ к°Җм§ҖлҠ” кі м§Ҳм Ғмқё л¬ём ңлҘј reactлҠ” 2018л…„ мғҲлЎңмҡҙ кё°лҠҘмқ„ нҶөн•ҳм—¬ нҡҚкё°м ҒмңјлЎң н•ҙкІ°н•ҳмҳҖлӢӨ.

# useState

stateкё°л°ҳмқҳ лқјмқҙн”„мӮ¬мқҙнҒҙмқ„ к°Җм§ҖлҠ” reactм—җм„ңлҠ” stateмқҳ кҙҖлҰ¬к°Җ л¬ҙм—Ү ліҙлӢӨ мӨ‘мҡ”н•ңлҚ°, мқҙлҘј нҺёлҰ¬н•ҳкІҢ м„ м–ён•ҳкі  мӮ¬мҡ©н•  мҲҳ мһҲлҸ„лЎқ н•ҙмӨҖлӢӨ.

кё°мЎҙмқҳ class componentм—җм„ңлҠ” constructorлӮҙм—җм„ңл§Ң stateлҘј м„ м–ён•  мҲҳ мһҲм—Ҳм§Җл§Ң, javascriptм—җм„ң ліҖмҲҳлҘј лӢӨлЈЁл“Ҝмқҙ м—¬лҹ¬ stateлҘј useStateлҘј нҶөн•ҙ м„ м–ён•  мҲҳ мһҲкі , к°Ғк°Ғмқҳ stateлҘј к°Ғк°Ғмқҳ set functionмқ„ нҶөн•ҳм—¬ updateн•  мҲҳ мһҲлӢӨ.

function count () {
  const [count, setCount] = useState(0);

  return (
    <div>
      <div onClick={()=> setCount(count + 1)}>
        {count}
      </div>
    </div>
  )
}

# useEffect

useEffecлҠ” class component м—җм„ң cDM,cDUкіј cWU мқ„ н•©міҗ лҶ“мқҖ кё°лҠҘмқ„ лӢҙлӢ№н•ңлӢӨ. мқҙлҘј нҶөн•ҙ мҡ°лҰ¬лҠ” useEffectлҘј нҶөн•ҳм—¬ lifecycleмқ„ нҶөн•ң мғҒнғңкҙҖлҰ¬ л°Ҹ л ҢлҚ”л§Ғмқ„ мҶҗмүҪкІҢ н•  мҲҳ мһҲлҸ„лЎқ лҗҳм—ҲлӢӨ.

function Effect () {
  useEffect ( () => {
    effect
    return () => {
      clearup
    }
  }, [])
}

reactлҠ” м–ҙл–»кІҢ cDM, cDU, cWUлҘј useEffectм—җ лӢҙм•ҳлҠ”к°Җ?

мҡ°лҰ¬лҠ” л‘җ лІҲм§ё мқёмһҗм—җ мЈјлӘ©н•ҙм•ј н•ңлӢӨ.

useEffect (() => {
  effect
  return () => {
    clearup
  }
}, []) // 2лІҲм§ё мқёмһҗлЎңлҠ” л°°м—ҙмқҙ л“Өм–ҙмҳ¬ мҲҳ мһҲлӢӨ.

2лІҲм§ё мқёмһҗлЎң л“Өм–ҙмҳӨлҠ” мҡ”мҶҢл“Өм—җ л”°лқј useEffectлҘј мӢӨн–үн•  кІғмқём§Җ л§җ кІғмқём§Җм—җ лҢҖн•ң кІ°м •мқ„н• мҲҳ мһҲлӢӨ.

  1. 2лІҲмЁ° мқёмһҗк°Җ м—Ҷмқ„ кІҪмҡ° - cDM, cDU, cWU м—җ н•ҙлӢ№н•ҳлҠ” кІҪмҡ° лӘЁл‘җ мӢӨн–ү

  2. 2лІҲм§ё мқёмһҗм—җ л№Ҳ л°°м—ҙ([])к°Җ л“Өм–ҙмҳ¬ кІҪмҡ° - cDM, cWUм—җ н•ҙлӢ№н•ҳлҠ” кІҪмҡ° мӢӨн–ү

  3. 2лІҲм§ё мқёмһҗмқё л°°м—ҙм—җ мҡ”мҶҢлЎң нҠ№м • sate or propsк°Җ л“Өм–ҙмҳ¬ кІҪмҡ° - cDM, cDU, cWU лҘј н•ҙлӢ№н•ҳлҠ” мҡ”мҶҢм—җ мў…мҶҚм ҒмңјлЎң мӢӨн–үн•ңлӢӨ.

мқҙл ҮкІҢ useStateмҷҖ useEffectл§ҢмңјлЎңлҸ„ кё°ліём Ғмқё лӢӨм–‘н•ң кё°лҠҘл“Өмқ„ л§Ңл“ңлҠ” кІғм—җлҠ” л¬ҙлҰ¬к°Җ м—ҶлӢӨ. н•ҳм§Җл§Ң м•Ңм•„л‘җл©ҙ нҺёлҰ¬н•ҳкі  м“°мһ„мғҲ мһҲлҠ” м¶”к°Җм Ғмқё hooksлҘј м•Ңм•„ліҙмһҗ

# useCallback

const memoizedCallback = useCallback(
  () => {
    doSomething(a, b);
  },
  [a, b],
);

н•ЁмҲҳмҷҖ к·ём—җ н•ҙлӢ№н•ҳлҠ” мқёмһҗлҘј лҜёлҰ¬ л§Ңл“Өм–ҙл‘ҳ н•„мҡ”м„ұмқҙ м–ҙл””м—җ мһҲлҠ”к°ҖлҘј ліҙл©ҙ мқҙл ҮкІҢ мў…мҶҚм„ұмқ„ мң м§Җн•ҳлҠ”лҚ° лҸ„мӣҖмқҙ лҗңлӢӨ.

мқҙл ҮкІҢ н•ҙл‘җл©ҙ useEffectлӮҙм—җ н•ЁмҲҳлҘј мһ‘м„ұн•  н•„мҡ”м„ұмқҙ м—Ҷм–ҙм§җмңјлЎң мҪ”л“ңк°Җ мўҖлҚ” м§ҒкҙҖм ҒмңјлЎң н•ҙлӢ№ м»ҙнҸ¬л„ҢнҠём—җм„ң м§Ғм ‘ н•ЁмҲҳлҘј мһ‘м„ұн•ҳм—¬ мқҙн•ҙн•ҳкё°м—җ мүҪкІҢ л§Ңл“ӨмҲҳ мһҲлӢӨ.

function ProductPage({ productId }) {
  // вң… лӘЁл“  л ҢлҚ”л§Ғм—җм„ң ліҖкІҪлҗҳм§Җ м•ҠлҸ„лЎқ useCallbackмңјлЎң лһҳн•‘
  const fetchProduct = useCallback(() => {
    // ... productIdлЎң л¬ҙм–ёк°ҖлҘј н•©лӢҲлӢӨ ...
  }, [productId]); // вң… лӘЁл“  useCallback мў…мҶҚм„ұмқҙ м§Җм •лҗ©лӢҲлӢӨ

  return <ProductDetails fetchProduct={fetchProduct} />;
}

function ProductDetails({ fetchProduct }) {
  useEffect(() => {
    fetchProduct();
  }, [fetchProduct]); // вң… лӘЁл“  useEffect мў…мҶҚм„ұмқҙ м§Җм •лҗ©лӢҲлӢӨ
  // ...
}

лҳҗн•ң к·ёлғҘ мҪңл°ұмқ„ л§Ңл“Өм–ҙм„ң мӮ¬мҡ©н•ҳкі  мӢ¶мқ„л•ҢлҸ„ мӮ¬мҡ©н•  мҲҳ мһҲлӢӨ.
