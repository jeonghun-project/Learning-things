# 1.1. classлһҖ?

к°қмІҙлҘј м°Қм–ҙлӮј мҲҳ мһҲлҠ” мІӯмӮ¬м§„(blueprint) classлҘј мқҙмҡ©н•ҙм„ң к°қмІҙлҘј л§Ңл“Өм–ҙлӮј мҲҳ мһҲкі , мқҙл ҮкІҢ л§Ңл“Өм–ҙм§„ к°қмІҙлҘј instanceлқјкі  м№ӯн•ңлӢӨ.

class Person {
  constructor(name) {
    this._name = name;
  }

  sayHello() {
    console.log(`Hello! ${this._name}`);
  }
}

const me = new Person("Lee");
me.sayHello(); // Hello! Lee

console.log(me instanceof Person); // true

// constructлҠ” classк°қмІҙм—җ л°ҳл“ңмӢң мһҲкі  лӘ…мӢңн•ҳм§Җ м•Ҡмқ„мӢң л№Ҳ к°қмІҙлҘј л§Ңл“Өм–ҙ лӮёлӢӨ.

# 1.2. instance

classлҘј нҶөн•ҙ prototype-baseмқҳ к°қмІҙлҘј н• лӢ№ н•  мҲҳ мһҲлӢӨ.

мқҙл•Ң newлқјлҠ” м—°мӮ°мһҗмҷҖ н•Ёкјҗ нҒҙлһҳмҠӨ мқҙлҰ„мқ„ нҳём¶ңн•ҳл©ҙ нҒҙлһҳмҠӨ мқёмҠӨн„ҙмҠӨ мҰү к°қмІҙк°Җ мғқм„ұлҗңлӢӨ.

class Foo {}

const foo = new Foo();

new м—°мӮ°мһҗлҘј мӮ¬мҡ©н•ҳм§Җ м•Ҡмңјл©ҙ Typeerrorк°Җ л°ңмғқн•ңлӢӨ.

newлҠ” instance нӮӨмӣҢл“ңмқҙлӢӨ. к°Ғк°Ғмқҳ мқёмҠӨн„ҙмҠӨлҠ” classмқҳ лӮҙмһҘ н•ЁмҲҳмқё mthodм—җ м ‘к·јн•  мҲҳ мһҲлӢӨ.

class Foo {}

const foo = Foo(); // TypeError: Class constructor Foo cannot be invoked without 'new'

# 1.3. constructor

classлҠ” мғқм„ұмһҗлҘј н•„мҡ”лЎңн•ҳлҠ”лҚ°, мқҙ мғқм„ұмһҗк°Җ constructor мқҙлӢӨ.

мқёмҠӨн„ҙмҠӨлҘј мғқм„ұн•ҳкі , *нҒҙлһҳмҠӨ н•„л“ңлҘј мҙҲкё°нҷ”н•ҳлҠ” нҠ№мҲҳн•ң method

# 1.3.1. нҒҙлһҳмҠӨ н•„л“ң

нҒҙлһҳмҠӨ лӮҙл¶Җмқҳ мәЎмҠҗнҷ”лҗң ліҖмҲҳ, мҰү thisм—җ м¶”к°Җн•ң property

// нҒҙлһҳмҠӨ м„ м–ёл¬ё
class Car {
  // constructor(мғқм„ұмһҗ). мқҙлҰ„мқ„ л°”кҝҖ мҲҳ м—ҶлӢӨ.

    // _modelмқҖ нҒҙлһҳмҠӨ н•„л“ңмқҙлӢӨ.
    this.model = model;

}

constructorлҠ” мҷёл¶Җм—җм„ң м–ём ңлӢӨ м ‘к·јн•ҳм—¬ к°’мқ„ мҲҳм •н•  мҲҳ мһҲлҠ” нҠ№м§•мқҙ мһҲлӢӨ.(public) 2020/02 нҳ„мһ¬лҠ” # лҘј мқҙмҡ©н•ҙм„ң publicFieldмҷҖ privateFieldлҘј кө¬л¶„ н•  мҲҳ мһҲлӢӨ

# 1.3.2. static

instance мҷҖлҠ” лі„к°ңлЎң нҒҙлһҳмҠӨм—җм„ң мһҗмЈј мӮ¬мҡ©н•ҳлҠ” method нҳ№мқҖ valueм—җ м ‘к·јн•ҳкі мһҗ н•  л•Ң

мҰү, instanceлҘј мғқм„ұн•ҳм§Җ м•Ҡкі  м ‘к·јн•  мҲҳ мһҲлҸ„лЎқ н•ҙмӨҖлӢӨ.

class Car {
  static public = "no is mine";
  constructor(name, model, owner) {
    this._name = name;
    this._model = model;
    this._owner = owner;
  }

  static isPublic() {
    console.log(Car.public);
  }
}

this.engine = engineмқҖ set engineмқ„ нҳём¶ңн•ҳкі  value === engineмңјлЎң л§Ңл“Өм–ҙ мӨҖлӢӨ. (мҪңмҠӨнғқмқҙ л„ҳм№ҳлҠ” кІғмқ„ л°©м§Җн•  н•„мҡ”к°Җ мһҲмқҢмңјлЎң кё°мЎҙмқҳ ліҖмҲҳмҷҖ лӢӨлҘё ліҖмҲҳлӘ…мқ„ мӮ¬мҡ©н•ңлӢӨ.)

# 3. мғҒмҶҚ (Inheritance)

classлҠ” мғҒмҶҚмқҙ к°ҖлҠҘн•ҳлӢӨ. мҪ”л“ң мһ¬мӮ¬мҡ©мқҳ кҙҖм җм—җм„ң нҳҒмӢ м Ғмқё кё°лҠҘмқ„ н•  мҲҳ мһҲлӢӨ.

# 3.1. extends

extends нӮӨмӣҢл“ңлҠ” л¶ҖлӘЁ нҒҙлһҳмҠӨ(base class)лҘј мғҒмҶҚл°ӣлҠ” мһҗмӢқ нҒҙлһҳмҠӨ(sub class)лҘј м •мқҳн•  л•Ң мӮ¬мҡ©н•ңлӢӨ

class Car {
  constructor(name, cost) {
    this.name = name;
    this.cost = cost;
  }

  getCar(pay) {
    this.pay = pay;
    return this.pay > this.cost ? true : false;
  }

  setOption() {
    return this.getCar(this.pay)
      ? this.pay - this.cost > 100
        ? "Full Option"
        : "nomal"
      : "isnot your";
  }
}

class OldCar extends Car {
  setOption() {
    return this.getCar(this.pay)
      ? this.pay - this.cost > 100
        ? "Dealer"
        : "your"
      : "isnot your";
  }
}

const oldcar = new OldCar("mini", 200);

console.log(oldcar.getCar(300)); // true; -- Inheritance
console.log(oldcar.setOption()); // 'your' -- Overrding

// oldcarлҠ” OldCar нҒҙлһҳмҠӨмқҳ мқёмҠӨн„ҙмҠӨмқҙлӢӨ.
console.log(oldcar instanceof OldCar); // true
// oldcarлҠ” Car нҒҙлһҳмҠӨмқҳ мқёмҠӨн„ҙмҠӨмқҙлӢӨ.
console.log(oldcar instanceof Car); // true

overriding - мғҒмң„ нҒҙлһҳмҠӨк°Җ к°Җм§Җкі  мһҲлҠ” л©”мҶҢл“ңлҘј н•ҳмң„ нҒҙлһҳмҠӨк°Җ мһ¬м •мқҳн•ҳм—¬ мӮ¬мҡ©н•ҳлҠ” л°©мӢқмқҙлӢӨ

# 3.2. super

л¶ҖлӘЁ classмқҳ constructor, method лҘј нҳём¶ңн•  л•Ң мӮ¬мҡ©н•  мҲҳ мһҲлӢӨ.

мһҗмӢқ constructor лӮҙл¶Җм—җм„ң superлҘј нҶөн•ҙ л¶ҖлӘЁ constructorлҘј нҳём¶ңн•ҳм§Җ м•Ҡмңјл©ҙ Error

why? мһҗмӢқ instance лҳҗн•ң л¶ҖлӘЁмқҳ constructorлҘј default лЎң мӮ¬мҡ©н•ҳкё°м—җ, мӨ‘ліө м„ м–ёмқҙ лҗңлӢӨ. superлҘј нҶөн•ҙ м°ёмЎ°н•ҙмЈјм–ҙм•јл§Ң н•ңлӢӨ.

class Car {
   . . .
}

class OldCar extends Car {
	constructor (name, cost) {
		super(name); // л¶ҖлӘЁмқҳ constructorлҘј нҳём¶ң мқёмҲҳлҘј н• лӢ№н•ңлӢӨ.
		this.cost = cost
	}

  setOption() {
		// superлЎң л¶ҖлӘЁ нҒҙлһҳмҠӨлҘј м°ёмЎ°н•ҳм—¬ мӮ¬мҡ©н•  мҲҳ мһҲлӢӨ.
    return super.getCar(this.pay) ? this.pay - this.cost > 100 ? "Dealer" : "your" : 'isnot your';
  }
}

const oldcar = new OldCar("mini", 200);

console.log(oldcar.getCar(300)); // true; -- Inheritance
console.log(oldcar.setOption()); // 'your' -- Overrding

// oldcarлҠ” OldCar нҒҙлһҳмҠӨмқҳ мқёмҠӨн„ҙмҠӨмқҙлӢӨ.
console.log(oldcar instanceof OldCar); // true
// oldcarлҠ” Car нҒҙлһҳмҠӨмқҳ мқёмҠӨн„ҙмҠӨмқҙлӢӨ.
console.log(oldcar instanceof Car);   // true
